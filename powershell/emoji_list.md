    $emojiPath = Join-Path $emojiPath "Unicode-Emojis.json"
    $json = Get-Content $emojiPath | Out-String | ConvertFrom-Json

    $json | % {
    "|" + (get-emoji -Name $_.Name) +
    "|" + $_.Name +
    "|" + $_.Code +
    "|" + (($_.code -replace "U[+]", "&amp;#") -replace " ",";") + ";" +
    "|" + $_.Annotations + "|"
    } | set-clipboard
